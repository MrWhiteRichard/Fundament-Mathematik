\begin{align}

\int e^{\lambda t} \cdot t^n dt &=
\frac{ e^{\lambda t} }{\lambda} \cdot t^n - \int \frac{ e^{\lambda t} }{\lambda} \cdot n t^{n-1} dt \\
&= \frac{ e^{\lambda t} }{\lambda} \cdot t^n - \frac{n}{\lambda} \int e^{\lambda t} \cdot t^{n-1} dt \\
&= \cdots \\
&= \frac{ e^{\lambda t} }{\lambda} \cdot t^n - \frac{n}{\lambda} \left( \cdots - \int e^{\lambda t} \cdot t^{n-n} dt \right) \\
&= \frac{ e^{\lambda t} }{\lambda} \cdot t^n - \frac{n}{\lambda} \left( \cdots - \frac{e^{\lambda t}}{\lambda} \right) \\

\end{align}
