\begin{exercise}[Exercise 3.9]

Suppose $\gamma = 0.9$ and the reward sequence is $R_1 = 2$ followed by an infinite sequence of $7$s.
What are $G_1$ and $G_0$?    

\end{exercise}

% --------------------------------------------------------------------------------

\begin{solution}

We first apply the definition of $G_t$.

\begin{align*}
    G_1
    \stackrel
    {
        \eqref{eq:2.15}
    }{=}
    \sum_{k=0}^\infty
        \gamma^k R_{k+2}
    =
    \sum_{k=0}^\infty
        (0.9)^k \cdot 7
    =
    7 \cdot \frac{1}{1 - 0.9}
    =
    70
\end{align*}

Then, we use the recursion \eqref{eq:2.11}.

\begin{align*}
    G_0
    \stackrel
    {
        \eqref{eq:2.11}
    }{=}
    R_1 + \gamma G_1 = 65
\end{align*}

\end{solution}
