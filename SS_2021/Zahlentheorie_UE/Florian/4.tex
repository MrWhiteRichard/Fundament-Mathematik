\begin{exercise}

Zeigen Sie, dass für alle $n \geq 1$ gilt:
$(n! + 1, (n+1)! + 1) = 1$.

\end{exercise}

% -------------------------------------------------------------------------------- %

\begin{solution}

\textbf{NOCH UNGELÖST!!!}

Wir bemerken zuerst, dass $((n+1)!, (n+1)! + 1) = 1$,
$(n!, n! + 1) = 1$ und $(n!, (n+1)!) = n!$.

\begin{align*}
  n = 1: (2, 3) = 1 \\
  n = 2: (3, 7) = 1 \\
  n = 3: (7, 25) = 1 \\
  n = 4: (25, 121) = 1 \\
  n = 5: (121, 721) = 1
\end{align*}

Sei $d := (n! + 1, (n+1)! + 1) = 1$. Dann existieren $x,y \in \Z$ mit

\begin{align*}
  x(n! + 1) + y((n+1)! + 1) = d.
\end{align*}

\end{solution}
