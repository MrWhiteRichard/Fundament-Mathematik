    y^{\prime} = \begin{pmatrix}

    \end{pmatrix}y, \qquad

      1 & 0 \\ 0 & 0
    \end{pmatrix}y(0)
    + \begin{pmatrix}
      -1 & 0 \\ 0 & 1
    \end{pmatrix}y(1)

      0 \\ 1
    \end{pmatrix}.
  \end{align*}

    1 & 0 \\ 0 & 1
  \end{pmatrix}y(0)
  + \begin{pmatrix}
    -1 & 0 \\ 0 & -1
  \end{pmatrix}y(\pi)
  = \begin{pmatrix}
    0 \\ 0
  \end{pmatrix}
  \end{align*}
  in Abhängigkeit von $\alpha \in \R$.
\end{enumerate}
