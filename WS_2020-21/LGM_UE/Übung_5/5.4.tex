\begin{exercise}[110]

Skizzieren Sie einen weiteren (halb-)formalen Beweis einer der Formeln aus den vorigen
beiden Beispielen.

\end{exercise}

% --------------------------------------------------------------------------------

\begin{solution}

Wir skizzieren den Beweis für
$\exists x [A(x) \land B(x)] \rightarrow [\exists x A(x)] \land [\exists x B(x)]$:\\
Wir beweisen stattdessen
$[\forall x \neg A(x)] \land [\forall x \neg B(x)] \rightarrow \forall x [\neg(A(x) \land B(x))] $ und schließen
dann mit Tautologien auf die eigentliche Formel.
Wir kennen bereits einen (halbformalen) Beweis von
$[\forall x \neg A(x)] \land [\forall x \neg B(x)] \rightarrow \forall x [\neg A(x) \land \neg B(x)]$.
Jetzt müssen wir nur noch die offensichtliche Tautologie
mit dem Distributivitäts-Axiom kombinieren und wir sind fertig:
\begin{flalign*}
  &\vdash [\forall x \neg A(x)] \land [\forall x \neg B(x)] \rightarrow \forall x [\neg A(x) \land \neg B(x)] \\
  &\vdash \forall x \underbrace{(\neg A(x) \land \neg B(x))}_{C} \rightarrow
  \underbrace{(\neg(A(x) \land B(x)))}_{D}
  & \text{(Allquantor vor Tautologie)}\\
  &\vdash [\forall x (C \rightarrow D)]
  \rightarrow [\forall x C \rightarrow \forall x D] & \text{(Distributivität)} \\
  &\vdash \forall x C \rightarrow \forall x D & \text{(MP)} \\
  &\vdash [\forall x \neg A(x)] \land [\forall x \neg B(x)] \rightarrow \forall x [\neg(A(x) \land B(x))]
  & \text{(Tautologie + MP)}
\end{flalign*}
\end{solution}

\begin{solution}

	Wir skizzieren den Beweis für
	$\exists x [A(x) \land B(x)] \rightarrow [\exists x A(x)] \land [\exists x B(x)]$:\\
	\begin{flalign*}
	&\vdash (\forall x \neg A(x)) \rightarrow  \neg A(y) 							           & \text{(Substitutionsaxiom)}\\
	&\vdash A(y) \rightarrow \exists x A(x)              							           & \text{(Tautologie \eqref{eq:taut3} + Existenzaxiom + MP)} \\
	&\vdash B(y) \rightarrow \exists x B(x)              							           & \text{(analog)} \\
	&\vdash [A(y) \land B(y)] \rightarrow [\exists x A(x)] \land [\exists x B(x)]              & \text{(Tautologie \eqref{eq:taut8} + MP)} \\
	&\vdash \exists y ´[A(y) \land B(y)] \rightarrow [\exists x A(x)] \land [\exists x B(x)]   & (\exists\text{-Einführung stark)} \\
	&\vdash [\forall y \neg P(y)] \rightarrow [\forall x \neg P(x)]                            & \text{(haben wir bereits bewiesen)} \\
	&\vdash [\exists x P(x)] \rightarrow [\exists y P(y)]                                      & \text{(Tautologien \eqref{eq:taut3} und \eqref{eq:taut4} + Existenzaxiom + MP)} \\
	&\vdash \exists x [A(x) \land B(x)] \rightarrow [\exists x A(x)] \land [\exists x B(x)]    & \text{(Tautologie \eqref{eq:taut4} + MP)}
	\end{flalign*}
	Tautologien
	\begin{align}
	(p \rightarrow q) \rightarrow ((r \rightarrow s) \rightarrow ((p \land r) \rightarrow (q \land s))) \label{eq:taut8}
	\end{align}
\end{solution}
