\begin{exercise}[36]

 Geben Sie eine *erfüllbare Menge von $_\infty$Klauseln an, die nicht erfüllbar ist.

\end{exercise}

% -------------------------------------------------------------------------------- %

\begin{solution}

\begin{align*}
\Bbraces{\bigcup_{n=1}^\infty \{p_n\}} \cup \Bbraces{\{\neg p_n\} \mid n \in \Z^+}
\end{align*}

\end{solution}
