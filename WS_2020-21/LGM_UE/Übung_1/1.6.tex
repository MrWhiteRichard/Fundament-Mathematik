\begin{exercise}[19]

Zeigen Sie:
$(p \land q) \lor (\neg p \land \neg q) \Leftrightarrow (p \to q) \land (q \to p)$.

\end{exercise}

% --------------------------------------------------------------------------------

\hline
\end{tabular}
