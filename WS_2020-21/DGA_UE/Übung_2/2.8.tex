\begin{exercise}

Zeigen Sie:

\begin{align*}
  f(n) := 2^{2^{\floorbraces{\log_2{(\log_2{n})}}}} = \Landau(n)
\end{align*}

und bestimmen Sie die grÃ¶ÃŸte Zahl $c > 0$, sodass

\begin{align*}
  2^{2^{\floorbraces{\log_2{(\log_2{n})}}}} = \Omega(n^c)
\end{align*}

\end{exercise}

% --------------------------------------------------------------------------------

\begin{solution}

\phantom{}

\begin{enumerate}

  \item

  \begin{align*}
    \implies
    \Forall n \in \N:
    2^{2^{\floorbraces{\log_2{(\log_2{n})}}}}
    \leq
    2^{2^{\log_2{(\log_2{n})}}}
    =
    2^{(\log_2{n})}
    =
    n
    =
    \Landau(n)
  \end{align*}

  \item Wir behaupten, dass
  
  \begin{align*}
    1/2 = c := \sup{C},
    \quad
    C := \Bbraces{d > 0: 2^{2^{\floorbraces{\log_2{(\log_2{n})}}}} = \Omega(n^d)}.
  \end{align*}

  \begin{itemize}

    \item
    [\blockquote{$\leq$}:]

    \begin{align*}
      & \implies
      \Forall n \in \N:
      2^{2^{\floorbraces{\log_2{(\log_2{n})}}}}
      \geq
      2^{2^{\log_2{(\log_2{n})} - 1}}
      =
      2^{2^{\log_2{(\log_2{(n)})}} / 2}
      =
      (2^{\log_2{n}})^{1/2}
      =
      \sqrt{n}
      =
      \Omega(n^{1/2}) \\
      & \implies
      1/2 \in C
    \end{align*}

    \item
    [\blockquote{$\geq$}:]
    Sei $d > 1/2$, dann ist $1/2 - d < 0$.
    Betrachte die Folge $a_n := 2^{2^n} - 1$.

    \begin{align*}
      & \implies
      f(a_n) = 2^{2^{\floorbraces{\log_2{(\log_2{(a_n)})}}}}
      =
      \underbrace
      {
        2^{2^{\floorbraces{\log_2 \pbraces{\log_2 \pbraces{2^{2^n} - 1}}}}}
      }_{
        \stackrel{\text{scharf}}{<}
        2^{2^{\floorbraces{\log_2{\log_2{2^{2^n}}}}}}
        =
        2^{2^n}
      }
      =
      2^{2^{n-1}} \\
      & \implies
      \frac{f(a_n)}{a_n^d}
      =
      \frac{2^{2^{n-1}}}{(2^{2^n} - 1)^d}
      =
      \frac{(2^{2^n})^{1/2}}{(2^{2^n} - 1)^d}
      \leq
      \frac{(2^{2^n})^{1/2}}{(2^{2^n}/2)^d}
      \leq
      2^d \frac{(2^{2^n})^{1/2}}{(2^{2^n})^d}
      =
      2^d (2^{2^n})^{1/2 - d}
      \xrightarrow{n \to \infty}
      0 \\
      & \implies
      \liminf_{n \to \infty}
      \frac{f(n)}{n^d} = 0
      \iff
      f(n) \neq \Omega(n^d) \\
      & \implies
      d \not \in C
    \end{align*}

  \end{itemize}

\end{enumerate}

\end{solution}
