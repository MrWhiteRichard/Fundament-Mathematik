\begin{exercise}

\phantom{}

\begin{enumerate}[label = (\alph*)]

    \item Zeigen Sie mittels vollständiger Induktion, dass ein Algorithmus, dessen Laufzeit $T(n)$ (wobei $n = 2^k$, $k \in \Z^+$) der Rekursion

    \begin{align*}
        T(n)
        =
        \begin{cases}
            1                    & \text{für}~ n = 2 \\
            2 T(\frac{n}{2}) + 1 & \text{für}~ n = 2^k, k > 1
        \end{cases}
    \end{align*}

    genügt, $T(n) = n - 1$ erfüllt.

    \item Zeigen Sie mittels vollständiger Induktion, dass ein Algorithmus, dessen Laufzeit $T(n)$ (wobei $n = 2^k$, $k \in \Z^+$) der Rekursion

    \begin{align*}
        T(n)
        =
        \begin{cases}
            2                    & \text{für}~ n = 2 \\
            2 T(\frac{n}{2}) + n & \text{für}~ n = 2^k, k > 1
        \end{cases}
    \end{align*}

    genügt, $T(n) = n \log_2{(n)}$ erfüllt.

\end{enumerate}


\end{exercise}

% -------------------------------------------------------------------------------- %

\begin{solution}

\phantom{}

\begin{enumerate}[label = (\alph*)]

  \item Wir führen Induktion nach $k$ (und schreiben immer $2^k$ statt $n$).

  IA($k=1$):

  \begin{align*}
    \implies
    T(2^1)
    =
    T(2)
    =
    1
    =
    2 - 1
    =
    2^1 - 1
  \end{align*}

  IV: $T(2^{k-1}) = 2^{k-1} - 1$ \\
  IS($k-1 \mapsto k$):

  \begin{align*}
    \implies
    T(2^k)
    =
    2 T(2^{k-1}) + 1
    \stackrel{\text{IV}}{=}
    2 (2^{k-1} - 1) + 1
    =
    2^k - 2 + 1
    =
    2^k - 1
  \end{align*}

  \item Wir führen Induktion nach $k$ (und schreiben immer $2^k$ statt $n$).

  IA($k=1$):

  \begin{align*}
    \implies
    T(2^1)
    =
    T(2)
    =
    2
    =
    2^1 \log_2{2^1}
  \end{align*}

  IV: $T(2^{k-1}) = 2^{k-1} \log_2{2^{(k-1)}}$ \\
  IS($k-1 \mapsto k$):

  \begin{align*}
    \implies
    T(2^k)
    =
    2 T(2^{k-1}) + 2^k
    \stackrel{\text{IV}}{=}
    2 (2^{k-1} \log_2{2^{k-1}}) + 2^k
    =
    2^k(k-1) + 2^k
    =
    2^k k - 2^k + 2^k
    =
    2^k \log_2{(2^k)}
  \end{align*}

\end{enumerate}

\end{solution}
