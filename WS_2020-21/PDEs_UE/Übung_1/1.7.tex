\begin{exercise}

Gegeben sei das Anfangswertproblem

\begin{align*}
    u_t - u_{xx} = 0,
    \quad
    u(0, x) = g(x)
\end{align*}

f√ºr $u: [0, \infty)^2 \to \R$, wobei $g \in C^3(\R)$ ist.
Zeigen Sie, dass bei $t = 0$ die Funktionen $u_t, u_x, u_{xx}$ und $u_{tx}$ schon durch $g$ bestimmt sind.

\end{exercise}

% --------------------------------------------------------------------------------

\begin{solution}

\begin{align*}
    \implies
    g^\prime(x) = u_x(0, x)
    \implies
    \ExistsOnlyOne_{t=0} \enspace u_x
    \implies
    \ExistsOnlyOne_{t=0} \enspace u_{xx} = u_t
    \implies
    \ExistsOnlyOne_{t=0} \enspace u_{tx}
\end{align*}

\end{solution}
