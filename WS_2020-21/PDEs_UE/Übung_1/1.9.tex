\begin{enumerate}[label = (\roman*)]

    \item Zeigen Sie, dass für fixes $\lambda = \mu^2 > 0$ die Funktion

    \begin{align*}
        w(x, y) = (Ax + B)(C \cos{\mu y} + D \sin{\mu y})
    \end{align*}

    mit beliebigen Konstanten $A, B, C, D$ die homogene Helmholtz-Gleichung

    \begin{align*}
        \pderivative[2][w]{x}
        +
        \pderivative[2][w]{y}
        +
        \lambda w = 0
    \end{align*}

    löst.

    \item Zeigen Sie, dass die Funktion
    
    \begin{align*}
        w(x, y, z, t)
        =
        \frac{A}{(t - t_0)^{3/2}}
        \exp
        {
            \bbraces
            {
                -\frac
                {
                    (x - x_0)^2
                    +
                    (y - y_0)^2
                    +
                    (z - z_0)^2
                }{
                    4 a (t - t_0)
                }
            }
        }
    \end{align*}

    mit beliebigen Konstanten $A, t_0, x_0, y_0, z_0 \in \R$ die Wärmeleitungsgleichung

    \begin{align*}
        \pderivative[][w]{t}
        =
        a \pbraces
        {
            \pderivative[2][w]{x}
            +
            \pderivative[2][w]{y}
            +
            \pderivative[2][w]{z}
        }
    \end{align*}

    (mit $a \in \R$) löst.

\end{enumerate}

\end{exercise}
