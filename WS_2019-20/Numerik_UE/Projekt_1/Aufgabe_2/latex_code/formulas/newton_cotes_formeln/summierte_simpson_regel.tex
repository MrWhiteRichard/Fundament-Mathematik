    Q_h^{(2)}(f)
    & = \sum_{i=1}^n Q^{(2)}(f|_{[x_{i-1}, x_i]}) \\

        \pbraces
        {
            f(x_{i-1}) +
            4 f \pbraces{\frac{x_i + x_{i-1}}{2}} +

        } \\
    & = \frac{h}{6}
        \pbraces
        {
            \sum_{i=1}^n f(x_{i-1}) +

            \sum_{i=1}^n f(x_i)
        } \\
    & = \frac{1}{3}
        \pbraces
        {
            \frac{h}{2}
            \pbraces
            {
                \sum_{i=1}^n f(x_{i-1}) +
                \sum_{i=1}^n f(x_i)

        }
