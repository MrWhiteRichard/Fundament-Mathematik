\begin{align*}
    Q^{(1)}(f) \label{trapezium}
    & = \frac{b-a}{2}
        \pbraces
        {
            f(a) +
            f(b)
        } \\
    Q^{(2)}(f) \label{simpson}
    & = \frac{b-a}{6}
        \pbraces
        {
            f(a) +
            4 f \pbraces{\frac{a+b}{2}} +
            f(b)
        } \\
    Q^{(4)}(f) \label{mile}
    & = \frac{b-a}{90}
        \pbraces
        {
            7  f(a) + 
            32 f \pbraces{\frac{3a+b}{4}} +
            12 f \pbraces{\frac{a+b}{2}} +
            32 f \pbraces{\frac{a+3b}{4}} +
            7  f(b)
        }
\end{align*}
