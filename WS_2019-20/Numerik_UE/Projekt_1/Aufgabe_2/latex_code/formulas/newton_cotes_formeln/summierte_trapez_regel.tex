    & = \sum_{i=1}^n Q^{(1)}(f|_{[x_{i-1}, x_i]}) \\
    & = \sum_{i=1}^n
        \frac{x_i - x_{i-1}}{2}
        \pbraces
        {
            f(x_{i-1}) +
            f(x_i)
        } \\
    & = \frac{h}{2}
        \pbraces
        {
            \sum_{i=1}^n f(x_{i-1}) +
            \sum_{i=1}^n f(x_i)
        } \\
    & = \frac{h}{2}
        \pbraces
        {
            f(x_0) +
            \sum_{i=2}^n f(x_{i-1}) +
            \sum_{i=1}^{n-1} f(x_i) +
            f(x_n)
        } \\
    & = \frac{h}{2}
        \pbraces
        {
            f(a) +
            2 \sum_{i=1}^{n-1} f(x_i) +
            f(b)
        } \\
\end{align*}
