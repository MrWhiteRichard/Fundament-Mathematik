\begin{proof}

    v^{(1)}_2 & v^{(2)}_2 & \cdots & v^{(n-1)}_2 & \delta_{2 i} \\
    \vdots    & \vdots    & \ddots & \vdots      & \vdots \\
    v^{(1)}_n & v^{(2)}_n & \cdots & v^{(n-1)}_n & \delta_{n i}
  \end{vmatrix},

  \det{B_j}
  =
  \begin{vmatrix}
    v^{(1)}_1 & v^{(2)}_1 & \cdots & v^{(n-1)}_1 & v^{(j)}_1 \\

    \vdots    & \vdots    & \ddots & \vdots      & \vdots \\

  \sum_{i=1}^n
  (-1)^{n + i} \det{(A_i)}

  =
  \det{B_j} = 0
\end{align*}

\end{proof}
