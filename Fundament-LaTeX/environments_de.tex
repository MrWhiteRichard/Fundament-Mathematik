    colback = white,
    coltitle = black,
    fonttitle = \bfseries,
    sharp corners,
    detach title,
    width = 12cm,
    #1,
    code = {\ifdefempty{\tcbtitletext}{}{\tcbset{before upper = {{\centering \tcbtitle \par} \medskip}}}},
    boxrule = 0.5pt
}
